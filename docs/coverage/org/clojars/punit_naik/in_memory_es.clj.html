001  (ns org.clojars.punit-naik.in-memory-es
002    (:require [clojure.java.io :as io])
003    (:import [java.util HashMap Map]
004             [org.elasticsearch.common.settings Settings]
005             [org.elasticsearch.node Node]
006             [org.elasticsearch.node NodeBuilder]))
007  
008  (defn make-tmp-dir!
009    [cluster-name dir-name]
010    (let [base-dir (System/getProperty "java.io.tmpdir")
011          base-name (str (System/currentTimeMillis) "-" (long (rand 1000000000)))
012          custom-name (str cluster-name "-" dir-name)
013          tmp-dir (io/file (str base-dir "/" base-name "-" custom-name))]
014      (.mkdir tmp-dir)
015      (str tmp-dir)))
016  
017  (defn es-cluster-config
018    [{:keys [http-port tmp-data-dir tmp-logs-dir cluster-name tmp-home-dir]}]
019    (-> (Settings/builder)
020        (.put "http.port" http-port)
021        (.put "path.data" tmp-data-dir)
022        (.put "path.logs" tmp-logs-dir)
023        (.put "path.home" tmp-home-dir)
024        (.put "cluster.name" cluster-name)
025        (.put "index.number_of_shards" 1)
026        (.put "index.number_of_replicas" 0)
027        (.put "node.local" "true")
028        .build))
029  
030  (defn create-es-node
031    [^Settings settings]
032    (try (-> (NodeBuilder.)
033             (.settings settings)
034             .node)
035         (catch Exception e (.getMessage e))))
036  
037  (defn start-es-cluster
038    [^Settings settings]
039    (loop [node (create-es-node settings)]
040      (if (instance? Node node)
041        ^Node node
042        (let [http-port (Integer/parseInt (.get settings "http.port"))
043              ^Map settings-map (HashMap. (.getAsMap settings))]
044          (.put settings-map "http.port" (str (inc http-port)))
045          (recur
046           (create-es-node
047            (-> (Settings/builder)
048                (.put settings-map)
049                .build)))))))
050  
051  (defn start
052    [& [cluster-name http-port]]
053    (let [cluster-name (or cluster-name "test-cluster")
054          http-port (or http-port 9200)
055          tmp-data-dir (make-tmp-dir! cluster-name "data")
056          tmp-logs-dir (make-tmp-dir! cluster-name "logs")
057          tmp-home-dir (make-tmp-dir! cluster-name "home")
058          settings (es-cluster-config
059                    {:cluster-name cluster-name
060                     :http-port http-port
061                     :tmp-data-dir tmp-data-dir
062                     :tmp-logs-dir tmp-logs-dir
063                     :tmp-home-dir tmp-home-dir})
064          cluster (start-es-cluster settings)]
065      cluster))
066  
067  (defn stop
068    [^Node cluster]
069    (.close cluster))
